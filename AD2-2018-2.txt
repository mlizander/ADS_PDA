inicio
	numJogos <- 12.0
	jogoSemFazerGol <- 0
	numVitorias <- 0
	numEmpates <- 0
	saldoDeGols <- 0
	totalDeGols <- 0

	para i <- 1 até numJogos faça
		imprima("Jogo ", i, "?")
		leia(golsMarcados[i], golsSofridos[i])

		se golsMarcados[i] = 0 então
			jogoSemFazerGol <- jogoSemFazerGol + 1
		fim se
		
		se golsMarcados[i] > golsSofridos[i] então
			numVitorias <- numVitorias + 1
		fim se

		se golsMarcados[i] = golsSofridos[i] então
			numEmpates <- numEmpates + 1
		fim se
		saldoDeGols <- saldoDeGols + golsMarcados[i] - golsSofridos[i]
		totalDeGols <- totalDeGols + golsMarcados[i]
	próximo i

	mediaGols <- totalDeGols / numJogos
	totalPontos <- numVitorias*3 + numEmpates

	imprima("Pontos: ", totalPontos)
	imprima("Saldo de gols: ", saldoDeGols)
	imprima("Jogos sem fazer gols: ", jogoSemFazerGol)
	imprima("Média de gols por jogo: ", mediaGols)
fim

função trocaPares(entradas: frase)
inicio
	resultado <- frase
	posicao <- 1
	enquanto posicao < tamanho(frase)  faça
		se posicao < tamanho(frase) -1 então
			aux <- frase[posicao]
			frase[posicao] <- frase[posicao+1]
			frase[posicao+1] <- aux
		fim se
		posicao <- posicao + 2
	fim enquanto
fim
